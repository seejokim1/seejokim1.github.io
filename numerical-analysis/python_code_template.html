수치 적분 — Simpson's 1/3 Rule
Python simpson.py 
import numpy as np
import matplotlib.pyplot as plt

def simpsons_rule(f, a, b, n):
    """
    Simpson's 1/3 Rule for numerical integration
    Parameters:
        f : integrand function
        a : lower limit
        b : upper limit
        n : number of subintervals (must be even)
    """
    if n % 2 != 0:
        raise ValueError("n must be even for Simpson's rule")

    h      = (b - a) / n
    x      = np.linspace(a, b, n + 1)
    y      = f(x)

    # Simpson's coefficients: 1, 4, 2, 4, ..., 4, 1
    coeffs = np.ones(n + 1)
    coeffs[1:-1:2] = 4   # odd  indices
    coeffs[2:-2:2] = 2   # even indices

    return (h / 3) * np.dot(coeffs, y)

# --- Test: integral of sin(x) from 0 to pi ---
f      = lambda x: np.sin(x)
result = simpsons_rule(f, 0, np.pi, 100)
exact  = 2.0

print(f"Numerical : {result:.10f}")
print(f"Exact     : {exact:.10f}")
print(f"Error     : {abs(result - exact):.2e}")

# --- Visualization ---
x_plot = np.linspace(0, np.pi, 300)
plt.figure(figsize=(8, 4))
plt.fill_between(x_plot, f(x_plot), alpha=0.3, label='Area')
plt.plot(x_plot, f(x_plot), 'b-', lw=2, label=r'$\sin(x)$')

plt.title("Simpson's Rule Integration")

plt.tight_layout();  plt.show()
ODE — Euler vs Runge-Kutta 4th Order
Python ode_euler_rk4.py 
import numpy as np
import matplotlib.pyplot as plt

# ODE: dy/dx = -2y,  y(0)=1,  exact: y = exp(-2x)
def f(x, y):
    return -2 * y

def euler(f, x0, y0, x_end, h):
    """Forward Euler method"""
    xs, ys = [x0], [y0]
    while xs[-1] < x_end - 1e-12:
        xs.append(xs[-1] + h)
        ys.append(ys[-1] + h * f(xs[-2], ys[-2]))
    return np.array(xs), np.array(ys)

def rk4(f, x0, y0, x_end, h):
    """4th-order Runge-Kutta method"""
    xs, ys = [x0], [y0]
    while xs[-1] < x_end - 1e-12:
        x, y = xs[-1], ys[-1]
        k1 = f(x,         y)
        k2 = f(x + h/2,  y + h/2 * k1)
        k3 = f(x + h/2,  y + h/2 * k2)
        k4 = f(x + h,     y + h   * k3)
        xs.append(x + h)
        ys.append(y + h/6 * (k1 + 2*k2 + 2*k3 + k4))
    return np.array(xs), np.array(ys)

h = 0.2
xe, ye = euler(f, 0, 1, 2, h)
xr, yr = rk4  (f, 0, 1, 2, h)
x_ex   = np.linspace(0, 2, 200)

print(f"{'x':>5} {'Euler':>10} {'RK4':>10} {'Exact':>10}")
print("-" * 40)
for i in range(len(xe)):
    ex = np.exp(-2 * xe[i])
    print(ff"{xe[i]:>5.2f} {ye[i]:>10.6f} {yr[i]:>10.6f} {ex:>10.6f}")

plt.figure(figsize=(9, 4))
plt.plot(x_ex, np.exp(-2*x_ex), 'k-',   lw=2, label='Exact')
plt.plot(xe,   ye,               'r--o', ms=5, label='Euler')
plt.plot(xr,   yr,               'b--s', ms=5, label='RK4')
plt.xlabel('x');  plt.ylabel('y')
plt.title(r"ODE $dy/dx=-2y$ --- Euler vs RK4 ($h=0.2$)")
plt.legend();  plt.grid(True, alpha=0.3)

선형 방정식 — Gaussian Elimination (Partial Pivoting)
Python gaussian.py 
import numpy as np

def gaussian_elimination(A, b):
    """Solve Ax = b using Gaussian elimination + partial pivoting."""
    n  = len(b)
    Ab = np.hstack([A.astype(float),
                    b.reshape(-1, 1).astype(float)])

    for col in range(n):
        # Partial Pivoting
        max_row = col + np.argmax(abs(Ab[col:, col]))
        Ab[[col, max_row]] = Ab[[max_row, col]]

        if abs(Ab[col, col]) < 1e-12:
            raise ValueError("Singular matrix")

        # Forward Elimination
        for row in range(col + 1, n):
            fac        = Ab[row, col] / Ab[col, col]
            Ab[row, :] -= fac * Ab[col, :]

    # Back Substitution
    x = np.zeros(n)
    for i in range(n - 1, -1, -1):
        x[i] = (Ab[i, -1] - np.dot(Ab[i, i+1:n], x[i+1:])) / Ab[i, i]
    return x

# --- Test ---
A = np.array([[ 2,  1, -1],
              [-3, -1,  2],
              [-2,  1,  2]], dtype=float)
b = np.array([8, -11, -3], dtype=float)

x     = gaussian_elimination(A, b)
x_ref = np.linalg.solve(A, b)

print(f"Solution (custom) : {x}")
print(f"Solution (numpy)  : {x_ref}")
print(f"||Ax - b||        : {np.linalg.norm(A @ x - b):.2e}")
